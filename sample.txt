CREATE TABLE public.bookings (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  user_id uuid NOT NULL,
  vehicle_id uuid,
  location text NOT NULL,
  scheduled_time timestamp with time zone NOT NULL,
  status text DEFAULT 'pending'::text CHECK (status = ANY (ARRAY['pending'::text, 'pending_payment'::text, 'assigned'::text, 'dispatched'::text, 'enroute'::text, 'arrived'::text, 'charging'::text, 'completed'::text, 'cancelled'::text])),
  estimated_price numeric,
  energy_requested text,
  created_at timestamp with time zone DEFAULT now(),
  charging_started_at timestamp with time zone,
  completed_at timestamp with time zone,
  assigned_at timestamp with time zone,
  payment_locked boolean DEFAULT false,
  cancellation_allowed boolean DEFAULT true,
  cancellation_reason text,
  payment_status text,
  assignment_reason text,
  rejection_reason text,
  latitude double precision,
  longitude double precision,
  assigned_van_id uuid,
  assigned_depot_id uuid,
  dispatch_status text DEFAULT 'pending'::text,
  eta_minutes double precision,
  live_tracking_enabled boolean DEFAULT false,
  assignment_score double precision,
  CONSTRAINT bookings_pkey PRIMARY KEY (id),
  CONSTRAINT bookings_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id),
  CONSTRAINT bookings_vehicle_id_fkey FOREIGN KEY (vehicle_id) REFERENCES public.vehicles(id),
  CONSTRAINT bookings_assigned_van_id_fkey FOREIGN KEY (assigned_van_id) REFERENCES public.charging_vans(id),
  CONSTRAINT bookings_assigned_depot_id_fkey FOREIGN KEY (assigned_depot_id) REFERENCES public.depots(id)
);
CREATE TABLE public.charging_vans (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  depot_id uuid,
  van_name text NOT NULL,
  current_latitude double precision,
  current_longitude double precision,
  current_status text DEFAULT 'available'::text,
  supported_vehicle_types ARRAY DEFAULT '{2W,3W,4W}'::text[],
  current_energy_level double precision DEFAULT 100.0,
  thermal_status text DEFAULT 'normal'::text,
  safety_status text DEFAULT 'ok'::text,
  telemetry_status text DEFAULT 'online'::text,
  active_booking_id uuid,
  sessions_completed_today integer DEFAULT 0,
  max_sessions_per_day integer DEFAULT 12,
  last_seen_at timestamp with time zone DEFAULT timezone('utc'::text, now()),
  created_at timestamp with time zone DEFAULT timezone('utc'::text, now()),
  CONSTRAINT charging_vans_pkey PRIMARY KEY (id),
  CONSTRAINT charging_vans_depot_id_fkey FOREIGN KEY (depot_id) REFERENCES public.depots(id)
);
CREATE TABLE public.depots (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  name text NOT NULL,
  latitude double precision NOT NULL,
  longitude double precision NOT NULL,
  service_radius_km double precision NOT NULL DEFAULT 8.0,
  region text,
  is_active boolean DEFAULT true,
  created_at timestamp with time zone DEFAULT timezone('utc'::text, now()),
  CONSTRAINT depots_pkey PRIMARY KEY (id)
);

  id uuid NOT NULL DEFAULT gen_random_uuid(),

  payment_method text,

  razorpay_order_id text,
  razorpay_signature text,
  receipt_no text,
  transaction_id text,
  paid_at timestamp with time zone,
  payment_locked_at timestamp with time zone,
  gateway_response jsonb,

  refund_status text,
  CONSTRAINT payments_pkey PRIMARY KEY (id),

CREATE TABLE public.subscriptions (
  id uuid NOT NULL DEFAULT uuid_generate_v4(),
  user_id uuid NOT NULL,
  plan_type text NOT NULL CHECK (plan_type = ANY (ARRAY['addon_membership'::text, 'lite_plan'::text])),
  status text NOT NULL CHECK (status = ANY (ARRAY['active'::text, 'expired'::text, 'cancelled'::text, 'pending'::text])),
  amount numeric NOT NULL,
  started_at timestamp with time zone DEFAULT now(),
  expires_at timestamp with time zone,
  auto_renew boolean DEFAULT true,
  created_at timestamp with time zone DEFAULT now(),
  updated_at timestamp with time zone DEFAULT now(),

  CONSTRAINT subscriptions_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
